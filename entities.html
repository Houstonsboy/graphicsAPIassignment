Entities (Pac-Man + 4 Opponents)

Coordinate system

  • r, c: tile coordinates (row, col) in the maze grid.
  • x, y: pixel coordinates on the canvas derived from tile coords.
// Helpers (in drizzy.html):
// tileToPixelX(col) = col * TILE
// tileToPixelY(row) = row * TILE
// tile center:
// x = tileToPixelX(c) + TILE/2
// y = tileToPixelY(r) + TILE/2

Entity object shape

// One entity instance (Pac-Man or a ghost)
{
  id: string,                    // stable id (e.g. "pacman", "ghost1")
  kind: "pacman" | "ghost",      // type

  // Position/state (tile-based)
  r: number,                     // current tile row
  c: number,                     // current tile col
  dir: "up"|"down"|"left"|"right" | null,

  // Rendering (sprite)
  spriteSrc: string,            // placeholder JPEG path
  img: HTMLImageElement | null, // loaded image
  loaded: boolean,

  // How the sprite should fit within one tile
  spriteScale: number,          // e.g. 0.9 means draw at 0.9*TILE
  drawOffsetY: number,          // pixel offset to align "feet" to corridors
}
